A* Pathfinding Algorithm

A* finds the shortest path between two points by balancing actual travel cost and a heuristic estimate of remaining distance.
Author

Andreas Bogossian

Published

March 19, 2026

  • Finding the shortest path between two points on a grid
  • Brute-force explores every possible path — complexity grows as O(b^d) where b is the branching factor and d is the depth
  • Need a smarter way to prioritize which paths to explore

Cost Function

  • Every node n is assigned a score: f(n) = g(n) + h(n)
  • g(n) — exact cost of the path from start to n (known)
  • h(n) — estimated cost from n to the goal (heuristic — not exact)
  • A* always expands the node with the lowest f(n) first
  • If h(n) = 0, A* degenerates into Dijkstra’s algorithm (explores in all directions equally)
  • If h(n) never overestimates the true cost, it is called admissible and A* is guaranteed to find the optimal path

Open and Closed Sets

  • Open set — nodes discovered but not yet fully evaluated (candidates for the next step)
  • Closed set — nodes already evaluated (won’t be revisited)
  • At each step: pop the lowest f(n) node from the open set, evaluate its neighbours, add them to the open set if not already visited

Heuristic

  • The heuristic h(n) determines how “directed” the search is
  • Manhattan distance — h(n) = |x_1 - x_2| + |y_1 - y_2| — best for 4-directional grids
  • Euclidean distance — h(n) = \sqrt{(x_1 - x_2)^2 + (y_1 - y_2)^2} — straight-line distance, can underestimate on grids
  • Chebyshev distance — h(n) = \max(|x_1 - x_2|, |y_1 - y_2|) — best when diagonal moves are allowed
  • Weighted A* — f(n) = g(n) + w \cdot h(n) where w > 1 makes the search faster but potentially suboptimal
  • Stronger heuristics explore fewer nodes but require more assumptions about the search space

Demo

import { W, H, CELL, ROWS, COLS } from "./astar-constants.js"
import { appState }                from "./astar-state.js"
import { initSearch, stepOnce, resetSearch, heuristicFn } from "./astar-search.js"
import { generateMaze }            from "./astar-maze.js"
import { draw }                    from "./astar-draw.js"
viewof heuristic = Inputs.select(
  ["manhattan", "euclidean", "chebyshev", "zero (Dijkstra)"],
  { value: "manhattan", label: "Heuristic" }
)
// ── persistent state container ───────────────────────────────────────────────
app_persistent = ({ s: null })
// ── UI — reactive to heuristic ────────────────────────────────────────────────
app = {
  // initialise canvas, ctx, state and pointer listeners once
  if (!app_persistent.s) {
    const canvas = document.createElement("canvas");
    canvas.width = W; canvas.height = H;
    canvas.style.cssText = `display:block;width:100%;max-width:${W}px;cursor:crosshair`;
    Object.assign(app_persistent, {
      s:      appState(),
      canvas: canvas,
      ctx:    canvas.getContext("2d"),
    });

    // pointer / paint — attached once, reads s via app_persistent
    let painting = false, paintValue = true;
    function cellAt(e) {
      const rect = canvas.getBoundingClientRect(), scale = W / rect.width;
      return {
        r: Math.floor((e.clientY - rect.top)  * scale / CELL),
        c: Math.floor((e.clientX - rect.left) * scale / CELL),
      };
    }
    function applyEdit(r, c) {
      const s = app_persistent.s;
      if (r < 0 || r >= ROWS || c < 0 || c >= COLS) return;
      if ((r === s.start.r && c === s.start.c) || (r === s.end.r && c === s.end.c)) return;
      if (s.walls[r][c] === paintValue) return;
      s.walls[r][c] = paintValue;
      resetSearch(s); app_persistent.redraw();
    }
    canvas.addEventListener("pointerdown", e => {
      app_persistent.stopAnim();
      painting = true; canvas.setPointerCapture(e.pointerId);
      const {r, c} = cellAt(e);
      if (r < 0 || r >= ROWS || c < 0 || c >= COLS) return;
      const s = app_persistent.s;
      if ((r === s.start.r && c === s.start.c) || (r === s.end.r && c === s.end.c)) return;
      paintValue = !s.walls[r][c];
      applyEdit(r, c);
    });
    canvas.addEventListener("pointermove", e => {
      if (painting) { const {r, c} = cellAt(e); applyEdit(r, c); }
    });
    canvas.addEventListener("pointerup", () => { painting = false; });
  }

  const s      = app_persistent.s;
  const canvas = app_persistent.canvas;
  const ctx    = app_persistent.ctx;

  // ── react to heuristic input ───────────────────────────────────────────────
  if (s.heuristic !== heuristic) {
    s.heuristic = heuristic;
    if (s.animTimer) { clearTimeout(s.animTimer); s.animTimer = null; }
    s.animating = false;
    resetSearch(s);
  }

  // ── buttons ────────────────────────────────────────────────────────────────
  const btnAnimate = document.createElement("button");
  const btnReset   = document.createElement("button");
  const btnClear   = document.createElement("button");
  const btnMaze    = document.createElement("button");
  btnAnimate.textContent = "▶ Animate";
  btnReset.textContent   = "Reset";
  btnClear.textContent   = "Clear walls";
  btnMaze.textContent    = "Gen maze";
  const btnRow = document.createElement("div");
  btnRow.style.cssText = "display:flex;flex-wrap:wrap;gap:6px;margin-bottom:8px";
  btnRow.append(btnAnimate, btnReset, btnClear, btnMaze);

  // ── iterations display ─────────────────────────────────────────────────────
  const iterDisplay = document.createElement("div");
  iterDisplay.textContent = "iterations: —";

  const root = document.createElement("div");
  root.append(btnRow, canvas, iterDisplay);

  // ── helpers ────────────────────────────────────────────────────────────────
  function redraw() {
    draw(s, ctx);
    iterDisplay.textContent = s.done
      ? `iterations: ${s.iter} · ${s.status}`
      : s.iter > 0 ? `iterations: ${s.iter}…` : "iterations: —";
  }

  function stopAnim() {
    s.animating = false;
    if (s.animTimer) { clearTimeout(s.animTimer); s.animTimer = null; }
    btnAnimate.textContent = "▶ Animate";
  }

  // store on persistent so pointer listeners can call them
  app_persistent.redraw   = redraw;
  app_persistent.stopAnim = stopAnim;

  function animTick() {
    if (!s.animating) return;
    if (!s.running && !s.done) initSearch(s);
    stepOnce(s);
    redraw();
    if (!s.done) s.animTimer = setTimeout(animTick, 50);
    else         stopAnim();
  }

  function startAnim() {
    if (s.done) resetSearch(s);
    s.animating = true;
    btnAnimate.textContent = "⏸ Pause";
    animTick();
  }

  // ── button handlers ────────────────────────────────────────────────────────
  btnAnimate.onclick = () => s.animating ? stopAnim() : startAnim();
  btnReset.onclick   = () => { stopAnim(); resetSearch(s); redraw(); };
  btnClear.onclick   = () => {
    stopAnim();
    s.walls = Array.from({length: ROWS}, () => new Array(COLS).fill(false));
    resetSearch(s); redraw();
  };
  btnMaze.onclick  = () => { stopAnim(); generateMaze(s); redraw(); };

  redraw();
  return root;
}
